数据库表:
产品表(product)
{
id //产品编号
name //产品名称
unit //单位
price //价格
specification //规格
quality_remain //现在库存数量
remarks //备注
}

-- Describe PRODUCT
CREATE TABLE product (
    "id" TEXT PRIMARY KEY,
    "name" TEXT NOT NULL,
    "unit" TEXT NOT NULL,
    "price" REAL NOT NULL,
    "specification" TEXT NOT NULL,
    "quality_remain" REAL  NOT NULL  DEFAULT (0.0),
    "remarks" TEXT
)

订单表(order_list)
{
id // 订单编号
date //订单日期
sign_people //签收人
operator //操作人
remarks //备注
}
-- Describe ORDER_LIST
CREATE TABLE order_list (
    "id" TEXT PRIMARY KEY,
    "date" TEXT NOT NULL,
    "sign_people" TEXT NOT NULL,
    "operator" TEXT NOT NULL,
    "remarks" TEXT
)

出入库表(inout)
{
id // 主键 无意义
product_id // foreign key refer to table product
product_name // product name
order_number //订单号
date // date of the order
quality_change // 交易数量出库为负，入库为正
quality_remain //剩下的数量
abstract //摘要
}

-- Describe INOUT
CREATE TABLE inout (
    "id" INTEGER PRIMARY KEY autoincrement,
    "product_id" TEXT NOT NULL,
    "product_name" TEXT NOT NULL,
    "order_number" TEXT NOT NULL,
    "date" TEXT NOT NULL,
    "quality_change" INTEGER NOT NULL,
    "quality_remain" INTEGER NOT NULL,
    "abstract" TEXT,
     FOREIGN KEY(product_id) REFERENCES product(id),
     FOREIGN KEY(order_number) REFERENCES order_list(id)
)
